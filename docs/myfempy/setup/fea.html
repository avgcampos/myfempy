206    def Solve(self, solverset=None):
207        """
208        runSolve run the solver set
209
210        Arguments:
211            solvedata -- data information
212
213        Returns:
214            solution
215        """
216        print_console("solver")
217        # fulldofs = self.modelinfo['fulldofs']
218        # self.modelinfo = dict()
219        # self.modelinfo['coord'] = self.coord
220        # self.modelinfo['regions'] = self.regions
221        # loading_bar_v1(5,"SOLVER")
222        try:
223            solverset["solverstatus"] = dict()
224            self.symm = solverset["SYMM"]
225            if self.symm:
226                solverset["solverstatus"]["typeasmb"] = "SYMMETRIC"
227            else:
228                solverset["solverstatus"]["typeasmb"] = "FULL"
229        except:
230            self.symm = False
231            solverset["solverstatus"]["typeasmb"] = "FULL"
232        try:
233            self.mp = solverset["MP"]
234            solverset["solverstatus"]["ncpu"] = (
235                "PARALLEL_" + str(solverset["MP"]) + "_CORES"
236            )
237        except:
238            self.mp = 0
239            solverset["solverstatus"]["ncpu"] = "SERIAL_" + str(1) + "_CORE"
240        # loading_bar_v1(10,"SOLVER")
241        starttime = time()
242        assembly, forcelist = newAnalysis.Assembly(self, Model=self.model)
243        endttime = time()
244        solverset["solverstatus"]["timeasb"] = abs(endttime - starttime)
245        solverset["solverstatus"]["memorysize"] = (assembly["stiffness"].todense().nbytes)/1e6
246        # loading_bar_v1(50,"SOLVER")
247        try:
248            constrains = self.physic.constrains
249            freedof, fixedof, constdof = newAnalysis.getConstrains(self, constrains)
250            logging.info("TRY RUN CONSTRAINS -- SUCCESS")
251        except:
252            logging.warning("TRY RUN CONSTRAINS -- FAULT")
253        # loading_bar_v1(60,"SOLVER")
254        constrainsdof = dict()
255        constrainsdof["freedof"] = freedof
256        constrainsdof["fixedof"] = fixedof
257        constrainsdof["constdof"] = constdof
258        try:
259            Uc = newAnalysis.getDirichletNH(self, constrains)
260            logging.info("TRY SET DNH CONSTRAINS -- SUCCESS")
261        except:
262            logging.warning("TRY SET DNH CONSTRAINS -- FAULT")
263        nsteps = setSteps(solverset["STEPSET"])
264        if forcelist.shape[1] != nsteps:
265            forcelist = np.repeat(forcelist, nsteps, axis=1)
266        else:
267            pass
268        assembly["loads"] = forcelist
269        if Uc.shape[1] != nsteps:
270            Uc = np.repeat(Uc, nsteps, axis=1)
271        else:
272            pass
273        assembly["bcdirnh"] = Uc
274        # loading_bar_v1(80,"SOLVER")
275        try:
276            starttime = time()
277            solverset["solution"] = self.solver.runSolve(self.model, self.physic, assembly, constrainsdof, solverset)
278            endttime = time()
279            solverset["solverstatus"]["timesim"] = abs(endttime - starttime)
280            logging.info("TRY RUN SOLVER -- SUCCESS")
281        except:
282            logging.warning("TRY RUN SOLVER -- FAULT")
283        # loading_bar_v1(100,"SOLVER")
284        return solverset
