385    def Solve(self, solverset=None) -> dict:
386        """run the solver set
387
388        Keyword Arguments:
389            solverset -- configuration file for solution (default: {None})
390
391        Example:
392            ```python
393                solverset = {
394                    'STEPSET':{
395                        'type':'table',
396                        'start':0,
397                        'end':1,
398                        'step':1
399                    },
400                    'SYMM':False,
401                    'MP':False
402                }
403
404                solverdata = fea.Solve(solverset)
405            ```
406
407        Returns:
408            solverset
409        """
410        print_console("solver")
411        try:
412            solverset["solverstatus"] = dict()
413            self.symm = solverset["SYMM"]
414            if self.symm:
415                solverset["solverstatus"]["typeasmb"] = "SYMMETRIC"
416            else:
417                solverset["solverstatus"]["typeasmb"] = "FULL"
418        except:
419            self.symm = False
420            solverset["solverstatus"]["typeasmb"] = "FULL"
421        try:
422            self.mp = solverset["MP"]
423            solverset["solverstatus"]["ncpu"] = (
424                "PARALLEL_" + str(solverset["MP"]) + "_CORES"
425            )
426        except:
427            self.mp = 0
428            solverset["solverstatus"]["ncpu"] = "SERIAL_" + str(1) + "_CORE"
429        # loading_bar_v1(10,"SOLVER")
430        starttime = time()
431        assembly, forcelist = newAnalysis.Assembly(self, Model=self.model)
432        endttime = time()
433        solverset["solverstatus"]["timeasb"] = abs(endttime - starttime)
434        solverset["solverstatus"]["memorysize"] = (assembly["stiffness"].todense().nbytes)/1e6
435        # loading_bar_v1(50,"SOLVER")
436        try:
437            constrains = self.physic.constrains
438            freedof, fixedof, constdof = newAnalysis.getConstrains(self, constrains)
439            logging.info("TRY RUN CONSTRAINS -- SUCCESS")
440        except:
441            logging.warning("TRY RUN CONSTRAINS -- FAULT")
442        # loading_bar_v1(60,"SOLVER")
443        constrainsdof = dict()
444        constrainsdof["freedof"] = freedof
445        constrainsdof["fixedof"] = fixedof
446        constrainsdof["constdof"] = constdof
447        try:
448            Uc = newAnalysis.getDirichletNH(self, constrains)
449            logging.info("TRY SET DNH CONSTRAINS -- SUCCESS")
450        except:
451            logging.warning("TRY SET DNH CONSTRAINS -- FAULT")
452        nsteps = setSteps(solverset["STEPSET"])
453        if forcelist.shape[1] != nsteps:
454            forcelist = np.repeat(forcelist, nsteps, axis=1)
455        else:
456            pass
457        assembly["loads"] = forcelist
458        if Uc.shape[1] != nsteps:
459            Uc = np.repeat(Uc, nsteps, axis=1)
460        else:
461            pass
462        assembly["bcdirnh"] = Uc
463        # loading_bar_v1(80,"SOLVER")
464        try:
465            starttime = time()
466            solverset["solution"] = self.solver.runSolve(self.model, self.physic, assembly, constrainsdof, solverset)
467            endttime = time()
468            solverset["solverstatus"]["timesim"] = abs(endttime - starttime)
469            logging.info("TRY RUN SOLVER -- SUCCESS")
470        except:
471            logging.warning("TRY RUN SOLVER -- FAULT")
472        # loading_bar_v1(100,"SOLVER")
473        return solverset
